0004 Vector<int> //生成长度为n的随机有序向量，元素取值来自[min, max]
0005 randomSortedVector ( Rank n, int min, int max ) { //ACP, Vol.2, Algorithm S
0006    Vector<int> A;
0007    for ( int i = min; i <= max; i++ )
0008       if ( rand() % ( max - i + 1 ) < (int)n )
0009          { A.insert ( i ); n--; }
0010    return A;
0011 }
