0003 bool collide ( int* solu, int k ) { //判断是否可以在当前行的第k列放置下一皇后
0004    nCheck++;
0005    for ( int i = 0; i < k; i++ ) {
0006       if ( solu[k] == solu[i] )       return true; //列
0007       if ( solu[k] - solu[i] == k - i ) return true; //对角线
0008       if ( solu[k] - solu[i] == i - k ) return true; //反对角线
0009    }
0010    return false;
0011 }
