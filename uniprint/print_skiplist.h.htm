0005 template <typename K, typename V> //e、value
0006 void UniPrint::p ( Skiplist<K, V>& s ) { //引用
0007    printf ( "%s[%d]*%d*%d:\n", typeid ( s ).name(), (int) &s, s.height(), s.size() ); //基本信息
0008    s.traverse ( print ); //通过print()遍历输出所有元素
0009    printf ( "\n" );
0010 }
