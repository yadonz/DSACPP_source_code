0003 /******************************************************************************************
0004  * 显示当前各（窗口）队列情况
0005  ******************************************************************************************/
0006 void displayProgress ( Queue<Customer> windows[], Rank nWin, Rank now ) {
0007    system ( "cls" ); //清屏
0008    printf ( "============ Time: %4d ============\n", now );
0009    for ( Rank i = 0; i < nWin; i++ ) { //对每个窗口，分别
0010       printf ( "Window #%c:  ", 'A' + i ); //输出窗口编号
0011       windows[i].traverse ( printCustomer ); printf ( "\n" ); //列出队列中的所有顾客
0012    }
0013 }
