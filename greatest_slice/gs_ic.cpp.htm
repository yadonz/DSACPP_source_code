0001 extern int s_lo, s_hi;
0002 
0003 int gs_IC( int A[], int n ) { //递增策略：O(n^2)
0004    int gs = A[0]; s_lo = n; s_hi = -1;
0005    for ( int i = 0; i < n; i++ ) { //枚举所有起始于i
0006       int s = 0;
0007       for ( int j = i; j < n; j++ ) { //终止于j的区间
0008          s += A[j]; //递增地得到其总和：O(1)
0009          if ( gs < s ) { gs = s; s_lo = i; s_hi = j + 1; } //择优、更新
0010       }
0011    }
0012    return gs;
0013 }
