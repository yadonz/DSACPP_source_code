0002 /******************************************************************************************
0003  * 构造随机数组（可根据测试需要相应地改写）
0004  ******************************************************************************************/
0005 void randomArray ( int A[], int n, int seed ) {
0006    srand(seed);
0007    for (int i = 0; i < n; i++) A[i] = i;
0008    shuffle(A, n);
0009 }
