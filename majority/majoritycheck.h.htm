0001 template <typename T> bool majCheck ( Vector<T> A, T maj ) { //验证候选者是否确为众数
0002    Rank occurrence = 0; //maj在A[]中出现的次数
0003    for ( Rank i = 0; i < A.size(); i++ ) //逐一遍历A[]的各个元素
0004       if ( A[i] == maj ) occurrence++; //每遇到一次maj，均更新计数器
0005    return 2 * occurrence > A.size(); //根据最终的计数值，即可判断是否的确当选
0006 }
