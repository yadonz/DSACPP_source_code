0009 template <typename T> //向量的起泡排序（提前终止版）
0010 void Vector<T>::bubbleSort( Rank lo, Rank hi ) { //assert: 0 <= lo < hi <= size
0011    for( bool sorted = false; sorted = !sorted; hi-- )
0012       for( Rank i = lo+1; i < hi; i++ )
0013          if ( _elem[i - 1] > _elem[i] ) //若逆序，则
0014             swap( _elem[i - 1], _elem[i] ), sorted = false; //交换――因此不能判定已经整体有序
0015 }
