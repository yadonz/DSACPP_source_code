0001 #include "convert.h"
0002 
0003 /******************************************************************************************
0004  * 进制转换
0005  ******************************************************************************************/
0006 int main ( int argc, char* argv[] ) {
0007    for ( int i = 1; i < argc; i += 2 ) {
0008       __int64 n = _atoi64 ( argv[i] ); //待转换的十进制数
0009       int base = atoi ( argv[i+1] ); //目标进制
0010       Stack<char> S; //用栈记录转换得到的各数位
0011       convert ( S, n, base ); //进制转换
0012       while ( !S.empty() ) printf ( "%c", ( S.pop() ) ); //逆序输出栈内数位，即正确结果
0013    }
0014    return 0;
0015 }
