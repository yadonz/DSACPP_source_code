0011 template <typename T> void testQueue ( Rank n ) {
0012    Queue<T> Q;
0013    printf ( "\n  ==== Test %2d. Growing queue\n", testID++ );
0014    while ( Q.size() < n ) {
0015       ( Q.empty() || ( 30 < dice ( 100 ) ) ) ?
0016          Q.enqueue ( (T) dice ( 2 * n ) ) : //70%入队
0017          Q.dequeue(); //30%出队
0018       print ( Q );
0019    }
0020    printf ( "\n  ==== Test %2d. Shrinking queue\n", testID++ );
0021    while ( !Q.empty() ) {
0022       ( 70 < dice ( 100 ) ) ?
0023          Q.enqueue ( (T) dice ( 2 * n ) ) : //30%入队
0024          Q.dequeue(); //70%出队
0025       print ( Q );
0026    }
0027 }
